20template<typename T, T oHandler>
21struct callBackBuilder;
22
23template<typename Obj, typename... Arg, void(Obj::* oHandler)(Arg...)>
24struct callBackBuilder<void(Obj::*)(Arg...), oHandler>
25{
26 static void callback(Arg... arg)
27 {
28 int current_id = glutGetWindow();
29 //cout << "window [" << current_id << "] handler call" << endl;
30 auto it = anyWindow.find(current_id);
31 if (it != anyWindow.end())
32 {
33 (it->second->*oHandler)(arg...);
34 }
35 }
36};

89void MinGL::initGraphic()
90{
91 // Initialisation GLUT
92 glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGBA);
93
94 m_glutWindowId = glutCreateWindow(m_windowName.c_str());
95 anyWindow[m_glutWindowId] = this;
96
97 // On active la transparence
98 glEnable(GL_BLEND);
99 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
100
101 glutSetOption(GLUT_ACTION_ON_WINDOW_CLOSE, GLUT_ACTION_CONTINUE_EXECUTION);
102
103 // Initialisation handlers
104 glutReshapeFunc(BIND_CALLBACK(&MinGL::callReshape));
105 glutDisplayFunc(BIND_CALLBACK(&MinGL::callDisplay));
106 glutMouseFunc(BIND_CALLBACK(&MinGL::callMouse));
107 glutMotionFunc(BIND_CALLBACK(&MinGL::callMotion));
108 glutPassiveMotionFunc(BIND_CALLBACK(&MinGL::callPassiveMotion));
109 glutKeyboardFunc(BIND_CALLBACK(&MinGL::callKeyboard));
110 glutKeyboardUpFunc(BIND_CALLBACK(&MinGL::callKeyboardUp));
111 glutSpecialFunc(BIND_CALLBACK(&MinGL::callKeyboardSpecial));
112 glutSpecialUpFunc(BIND_CALLBACK(&MinGL::callKeyboardUpSpecial));
113 glutCloseFunc(BIND_CALLBACK(&MinGL::callClose));
114
115 // On set la couleur d'effacement (prend des float, donc obligé de diviser par la taille d'un GLuint)
116 glClearColor(m_bgColor.getRed() / 256.f, m_bgColor.getGreen() / 256.f, m_bgColor.getBlue() / 256.f, m_bgColor.getAlpha() / 256.f);
117
118 // Efface écran
119 clearScreen();
120
121 // Un tour de boucle pour lancer l'affichage
122 glutMainLoopEvent();
123 glutSetKeyRepeat(GLUT_KEY_REPEAT_OFF); // on désactive la répétition des touches
124 m_windowIsOpen = true;
125} // initGraphic()
