13# The "Simple Image for Space Invaders" (.si2) file format specifications
14# Authors: Alexandre "Kuruyia" SOLLIER, Marc AMBAUD (2019 - 2020)
15#
16# | Address | Length | Description |
17# |---------|--------|-------------------------------------|
18# | 0x00 | 2 | Magic number ("SI") |
19# | 0x02 | 4 | Header magic number ("HEAD") |
20# | 0x06 | 2 | File version |
21# | 0x08 | 4 | Pixel count |
22# | 0x0C | 4 | Image width |
23# | 0x10 | 4 | Data magic number ("DATA") |
24# | 0x14 | ??? | Uncompressed image data as RGBA8888 |
25# "File version", "Pixel count" and "Image width" entries are little-endian, unsigned numbers.

59def convert_to_si2(source, output):
60 # Print summary
61 _, outExtension = os.path.splitext(source)
62 print('Source image: {}'.format(outExtension))
63 print('Output image: .si2')
64
65 # Read source image
66 with Image.open(source) as img:

70 # Convert image mode if unsupported

74 imageData = imgConvt.load()
75 else:
76 imageData = img.load()
77
78 # Print summary 2: Electric Boogaloo
79 print('Image size is {}x{}'.format(imageSize[0], imageSize[1]))
80 print()
81 print('Starting conversion...')
82
83 # Convert the image
84 with open(output, 'wb') as file:
85 file.write(b'SIHEAD\01\00')
86 file.write((imageSize[0]*imageSize[1]).to_bytes(4, byteorder='little'))
87 file.write((imageSize[0]).to_bytes(4, byteorder='little'))
88 file.write(b'DATA')

90 for y in range(0, imageSize[1]):
91 for x in range(0, imageSize[0]):
92 file.write(bytes(imageData[x, y]))
